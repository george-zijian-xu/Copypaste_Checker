// Program.cs
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Runtime.InteropServices;
using System.Text;
using System.Threading;

// Optional UIA - wrapped in try/catch so it won't break if unavailable
// Add reference to UIAutomationClient if you want richer info.
#if USE_UIA
using System.Windows.Automation;
#endif

internal static class Program
{
    private const int WM_CLIPBOARDUPDATE = 0x031D;
    private const int WM_DESTROY = 0x0002;
    private const uint CF_UNICODETEXT = 13;

    // Win32 P/Invoke
    [DllImport("user32.dll", CharSet = CharSet.Unicode, SetLastError = true)]
    private static extern ushort RegisterClassExW(in WNDCLASSEX lpwcx);

    [DllImport("user32.dll", CharSet = CharSet.Unicode, SetLastError = true)]
    private static extern IntPtr CreateWindowExW(
        uint dwExStyle, string lpClassName, string lpWindowName, uint dwStyle,
        int X, int Y, int nWidth, int nHeight,
        IntPtr hWndParent, IntPtr hMenu, IntPtr hInstance, IntPtr lpParam);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern bool DestroyWindow(IntPtr hWnd);

    [DllImport("user32.dll")]
    private static extern IntPtr DefWindowProcW(IntPtr hWnd, uint msg, UIntPtr wParam, IntPtr lParam);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern sbyte GetMessageW(out MSG lpMsg, IntPtr hWnd, uint wMsgFilterMin, uint wMsgFilterMax);

    [DllImport("user32.dll")]
    private static extern bool TranslateMessage(in MSG lpMsg);

    [DllImport("user32.dll")]
    private static extern IntPtr DispatchMessageW(in MSG lpMsg);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern bool AddClipboardFormatListener(IntPtr hwnd);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern bool RemoveClipboardFormatListener(IntPtr hwnd);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern bool OpenClipboard(IntPtr hWndNewOwner);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern bool CloseClipboard();

    [DllImport("user32.dll")]
    private static extern IntPtr GetClipboardOwner();

    [DllImport("user32.dll")]
    private static extern uint GetWindowThreadProcessId(IntPtr hWnd, out uint lpdwProcessId);

    [DllImport("kernel32.dll", SetLastError = true)]
    private static extern IntPtr GlobalLock(IntPtr hMem);

    [DllImport("kernel32.dll", SetLastError = true)]
    private static extern bool GlobalUnlock(IntPtr hMem);

    [DllImport("user32.dll", SetLastError = true)]
    private static extern IntPtr GetClipboardData(uint uFormat);

    [DllImport("user32.dll", SetLastError = true, CharSet = CharSet.Unicode)]
    private static extern ushort RegisterClipboardFormatW(string lpszFormat);

    [DllImport("user32.dll")]
    private static extern uint EnumClipboardFormats(uint format);

    [DllImport("user32.dll", SetLastError = true, CharSet = CharSet.Unicode)]
    private static extern int GetClipboardFormatNameW(uint format, StringBuilder lpszFormatName, int cchMaxCount);

    [DllImport("user32.dll")]
    private static extern uint GetClipboardSequenceNumber();

    [DllImport("user32.dll")]
    private static extern IntPtr GetForegroundWindow();

    [DllImport("user32.dll", CharSet = CharSet.Unicode)]
    private static extern int GetWindowTextW(IntPtr hWnd, StringBuilder lpString, int nMaxCount);

    [DllImport("kernel32.dll")]
    private static extern IntPtr GetModuleHandleW(string lpModuleName);

    [DllImport("user32.dll")]
    private static extern void PostQuitMessage(int nExitCode);

    // simple Win32 structs
    [StructLayout(LayoutKind.Sequential, CharSet = CharSet.Unicode)]
    private struct WNDCLASSEX
    {
        public uint cbSize;
        public uint style;
        public IntPtr lpfnWndProc;
        public int cbClsExtra;
        public int cbWndExtra;
        public IntPtr hInstance;
        public IntPtr hIcon;
        public IntPtr hCursor;
        public IntPtr hbrBackground;
        [MarshalAs(UnmanagedType.LPWStr)] public string lpszMenuName;
        [MarshalAs(UnmanagedType.LPWStr)] public string lpszClassName;
        public IntPtr hIconSm;
    }

    [StructLayout(LayoutKind.Sequential)]
    private struct MSG
    {
        public IntPtr hwnd;
        public uint message;
        public UIntPtr wParam;
        public IntPtr lParam;
        public uint time;
        public int pt_x;
        public int pt_y;
        public uint lPrivate;
    }

    // Delegate keepalive
    private static WndProcDelegate _wndProcDelegate;
    private delegate IntPtr WndProcDelegate(IntPtr hWnd, uint msg, UIntPtr wParam, IntPtr lParam);

    [STAThread]
    private static void Main()
    {
        Console.OutputEncoding = Encoding.UTF8;
        Console.WriteLine("Clipboard deep-dump listener (EXE). Ctrl+C to exit.");

        _wndProcDelegate = WndProc;

        var wc = new WNDCLASSEX
        {
            cbSize = (uint)Marshal.SizeOf<WNDCLASSEX>(),
            style = 0x0020, // CS_OWNDC
            lpfnWndProc = Marshal.GetFunctionPointerForDelegate(_wndProcDelegate),
            cbClsExtra = 0,
            cbWndExtra = 0,
            hInstance = GetModuleHandleW(null),
            hIcon = IntPtr.Zero,
            hCursor = IntPtr.Zero,
            hbrBackground = IntPtr.Zero,
            lpszMenuName = null,
            lpszClassName = "ClipboardDumpWindow",
            hIconSm = IntPtr.Zero
        };

        ushort atom = RegisterClassExW(in wc);
        if (atom == 0)
        {
            Console.WriteLine("RegisterClassExW failed: " + Marshal.GetLastWin32Error());
            return;
        }

        IntPtr hwnd = CreateWindowExW(
            0, wc.lpszClassName, "ClipboardDumpHidden", 0,
            0, 0, 0, 0, IntPtr.Zero, IntPtr.Zero, wc.hInstance, IntPtr.Zero);

        if (hwnd == IntPtr.Zero)
        {
            Console.WriteLine("CreateWindowExW failed: " + Marshal.GetLastWin32Error());
            return;
        }

        if (!AddClipboardFormatListener(hwnd))
        {
            Console.WriteLine("AddClipboardFormatListener failed: " + Marshal.GetLastWin32Error());
            DestroyWindow(hwnd);
            return;
        }

        Console.CancelKeyPress += (s, e) =>
        {
            Console.WriteLine("Shutting down...");
            RemoveClipboardFormatListener(hwnd);
            DestroyWindow(hwnd);
            // allow main to exit gracefully
        };

        // Message loop
        MSG msg;
        while (GetMessageW(out msg, IntPtr.Zero, 0, 0) > 0)
        {
            TranslateMessage(in msg);
            DispatchMessageW(in msg);
        }
    }

    private static IntPtr WndProc(IntPtr hWnd, uint msg, UIntPtr wParam, IntPtr lParam)
    {
        if (msg == WM_CLIPBOARDUPDATE)
        {
            try
            {
                HandleClipboardUpdate();
            }
            catch (Exception ex)
            {
                Console.WriteLine("Error handling clipboard update: " + ex);
            }
            return IntPtr.Zero;
        }
        else if (msg == WM_DESTROY)
        {
            RemoveClipboardFormatListener(hWnd);
            PostQuitMessage(0);
            return IntPtr.Zero;
        }

        return DefWindowProcW(hWnd, msg, wParam, lParam);
    }

    private static void HandleClipboardUpdate()
    {
        var timestamp = DateTime.UtcNow;
        uint seq = 0;
        try { seq = GetClipboardSequenceNumber(); } catch { }

        Console.WriteLine("\n---- CLIPBOARD UPDATE @ " + timestamp.ToString("o") + " (seq=" + seq + ") ----");

        // Owner window -> process
        try
        {
            IntPtr owner = GetClipboardOwner();
            if (owner != IntPtr.Zero)
            {
                uint pid;
                GetWindowThreadProcessId(owner, out pid);
                Console.WriteLine("Clipboard owner hwnd=" + owner + " pid=" + pid);
                TryPrintProcessInfo((int)pid);
                Console.WriteLine("Owner window title: " + GetWindowTextSafe(owner));
            }
            else
            {
                Console.WriteLine("Clipboard owner: <none>");
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine("Owner lookup failed: " + ex.Message);
        }

        // Foreground window
        try
        {
            IntPtr fg = GetForegroundWindow();
            uint fpid;
            GetWindowThreadProcessId(fg, out fpid);
            Console.WriteLine("Foreground hwnd=" + fg + " pid=" + fpid);
            TryPrintProcessInfo((int)fpid);
            Console.WriteLine("Foreground window title: " + GetWindowTextSafe(fg));
        }
        catch { }

        // Focused element via UIA (best-effort)
        try
        {
#if USE_UIA
            var focused = AutomationElement.FocusedElement;
            if (focused != null)
            {
                Console.WriteLine("Focused element: ControlType=" + focused.Current.ControlType.ProgrammaticName +
                    " Name=\"" + focused.Current.Name + "\" AutomationId=" + focused.Current.AutomationId);
                // try value pattern
                if (focused.TryGetCurrentPattern(ValuePattern.Pattern, out object vp))
                {
                    var v = (ValuePattern)vp;
                    Console.WriteLine("ValuePattern.Value (snippet): " + Snip(v.Current.Value, 200));
                }
                else if (focused.TryGetCurrentPattern(TextPattern.Pattern, out object tp))
                {
                    var t = (TextPattern)tp;
                    var r = t.DocumentRange;
                    Console.WriteLine("TextPattern.DocumentRange (snippet): " + Snip(r.GetText(-1), 300));
                }
            }
            else Console.WriteLine("Focused element: <null>");
#else
            Console.WriteLine("UIA not enabled in this build (define USE_UIA to enable).");
#endif
        }
        catch (Exception ex)
        {
            Console.WriteLine("UIA lookup failed: " + ex.Message);
        }

        // try open clipboard
        if (!TryOpenClipboard())
        {
            Console.WriteLine("OpenClipboard failed (busy).");
            return;
        }

        try
        {
            // enumerate formats
            Console.WriteLine("Available clipboard formats:");
            uint fmt = 0;
            var formats = new List<uint>();
            while (true)
            {
                fmt = EnumClipboardFormats(fmt);
                if (fmt == 0) break;
                formats.Add(fmt);
            }

            if (formats.Count == 0)
            {
                Console.WriteLine("  <none>");
            }
            else
            {
                foreach (var f in formats)
                {
                    string name = GetFormatName(f);
                    Console.WriteLine($"  fmt={f} name=\"{name}\"");
                    DumpFormatData(f);
                }
            }
        }
        finally
        {
            CloseClipboard();
        }
    }

    private static void DumpFormatData(uint format)
    {
        try
        {
            // for common types, special-case
            if (format == CF_UNICODETEXT)
            {
                var txt = GetClipboardUnicodeText();
                Console.WriteLine("    [CF_UNICODETEXT] len=" + (txt?.Length ?? 0));
                if (!string.IsNullOrEmpty(txt)) Console.WriteLine("    text: \"" + Snip(txt, 500) + "\"");
                return;
            }

            // try to get data pointer
            IntPtr h = GetClipboardData(format);
            if (h == IntPtr.Zero)
            {
                Console.WriteLine("    GetClipboardData returned NULL");
                return;
            }

            IntPtr p = IntPtr.Zero;
            try
            {
                p = GlobalLock(h);
                if (p == IntPtr.Zero)
                {
                    Console.WriteLine("    GlobalLock failed");
                    return;
                }

                // Try to interpret as text first (CF_HTML is often ANSI)
                string asAnsi = PtrToAnsiString(p, maxBytes: 16 * 1024);
                if (!string.IsNullOrEmpty(asAnsi))
                {
                    // print a header/snippet
                    Console.WriteLine("    asAnsi (snippet): " + Snip(asAnsi, 1000));
                    // If it's CF_HTML style, try to parse header
                    if (GetFormatName(format).IndexOf("HTML", StringComparison.OrdinalIgnoreCase) >= 0 ||
                        asAnsi.IndexOf("SourceURL:", StringComparison.OrdinalIgnoreCase) >= 0)
                    {
                        Console.WriteLine("    ---- CF_HTML header attempt ----");
                        string src = ExtractSourceUrlFromCFHtml(asAnsi);
                        if (src != null) Console.WriteLine("    SourceURL: " + src);
                        var hdrs = ParseCfHtmlHeaders(asAnsi);
                        foreach (var kv in hdrs) Console.WriteLine("    hdr: " + kv.Key + " = " + kv.Value);
                    }
                    return;
                }

                // fallback: hexdump first 512 bytes
                var hex = HexDump(p, 512);
                Console.WriteLine("    hex (first 512 bytes): " + hex);
            }
            finally
            {
                if (p != IntPtr.Zero) GlobalUnlock(h);
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine("    DumpFormatData failed: " + ex.Message);
        }
    }

    private static string GetFormatName(uint format)
    {
        try
        {
            // standard known constants
            switch (format)
            {
                case 1: return "CF_TEXT";
                case 2: return "CF_BITMAP";
                case 3: return "CF_METAFILEPICT";
                case 4: return "CF_SYLK";
                case 5: return "CF_DIF";
                case 6: return "CF_TIFF";
                case 7: return "CF_OEMTEXT";
                case 8: return "CF_DIB";
                case 9: return "CF_PALETTE";
                case 10: return "CF_PENDATA";
                case 11: return "CF_RIFF";
                case 12: return "CF_WAVE";
                case 13: return "CF_UNICODETEXT";
                case 14: return "CF_ENHMETAFILE";
                case 15: return "CF_HDROP";
                case 16: return "CF_LOCALE";
                case 17: return "CF_DIBV5";
            }

            // try to ask the system for name
            var sb = new StringBuilder(256);
            int r = GetClipboardFormatNameW(format, sb, sb.Capacity);
            if (r > 0) return sb.ToString();
            return "FORMAT_" + format;
        }
        catch { return "FORMAT_" + format; }
    }

    private static string GetClipboardUnicodeText()
    {
        IntPtr h = GetClipboardData(CF_UNICODETEXT);
        if (h == IntPtr.Zero) return null;
        IntPtr p = IntPtr.Zero;
        try
        {
            p = GlobalLock(h);
            if (p == IntPtr.Zero) return null;
            return Marshal.PtrToStringUni(p);
        }
        finally
        {
            if (p != IntPtr.Zero) GlobalUnlock(h);
        }
    }

    private static bool TryOpenClipboard()
    {
        if (OpenClipboard(IntPtr.Zero)) return true;
        Thread.Sleep(40);
        return OpenClipboard(IntPtr.Zero);
    }

    private static string PtrToAnsiString(IntPtr ptr, int maxBytes = 4096)
    {
        var bytes = new List<byte>();
        for (int i = 0; i < maxBytes; i++)
        {
            byte b = Marshal.ReadByte(ptr, i);
            if (b == 0) break;
            bytes.Add(b);
        }

        if (bytes.Count == 0) return null;
        // Many CF_HTML blob texts are ASCII/ANSI; try ASCII then UTF8 fallback
        try { return Encoding.ASCII.GetString(bytes.ToArray()); }
        catch { }
        try { return Encoding.UTF8.GetString(bytes.ToArray()); }
        catch { return null; }
    }

    private static string HexDump(IntPtr ptr, int maxBytes)
    {
        try
        {
            var sb = new StringBuilder();
            for (int i = 0; i < maxBytes; i++)
            {
                byte b = Marshal.ReadByte(ptr, i);
                sb.Append(b.ToString("x2"));
                if (i % 32 == 31) { sb.AppendLine(); }
                else sb.Append(' ');
            }
            return sb.ToString();
        }
        catch { return "<hexdump failed>"; }
    }

    private static string ExtractSourceUrlFromCFHtml(string cfHtml)
    {
        if (string.IsNullOrEmpty(cfHtml)) return null;
        const string marker = "SourceURL:";
        int idx = cfHtml.IndexOf(marker, StringComparison.OrdinalIgnoreCase);
        if (idx < 0) return null;
        int start = idx + marker.Length;
        int end = start;
        while (end < cfHtml.Length && cfHtml[end] != '\r' && cfHtml[end] != '\n') end++;
        string url = cfHtml.Substring(start, end - start).Trim();
        return string.IsNullOrEmpty(url) ? null : url;
    }

    private static Dictionary<string, string> ParseCfHtmlHeaders(string cfHtml)
    {
        var d = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
        if (string.IsNullOrEmpty(cfHtml)) return d;
        // read the first 1024 chars looking for header lines
        var snippet = cfHtml.Length > 4096 ? cfHtml.Substring(0, 4096) : cfHtml;
        var lines = snippet.Split(new[] { "\r\n", "\n" }, StringSplitOptions.None);
        foreach (var L in lines)
        {
            int k = L.IndexOf(':');
            if (k > 0 && k < 60)
            {
                string key = L.Substring(0, k).Trim();
                string val = L.Substring(k + 1).Trim();
                if (!d.ContainsKey(key)) d[key] = val;
            }
        }
        return d;
    }

    private static string Snip(string s, int max)
    {
        if (s == null) return "";
        if (s.Length <= max) return s;
        return s.Substring(0, max) + "...";
    }

    private static string GetWindowTextSafe(IntPtr hwnd)
    {
        try
        {
            var sb = new StringBuilder(1024);
            int r = GetWindowTextW(hwnd, sb, sb.Capacity);
            if (r > 0) return sb.ToString();
            return "<no-title>";
        }
        catch { return "<err>"; }
    }

    private static void TryPrintProcessInfo(int pid)
    {
        try
        {
            var proc = Process.GetProcessById(pid);
            Console.WriteLine($"    Process: Id={proc.Id} Name={proc.ProcessName} Path={SafeGetMainModuleFileName(proc)}");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"    Process lookup failed for pid={pid}: {ex.Message}");
        }
    }

    private static string SafeGetMainModuleFileName(Process p)
    {
        try { return p.MainModule?.FileName ?? p.ProcessName; }
        catch { try { return p.ProcessName; } catch { return "<unknown>"; } }
    }
}
